\section{Customer Satisfaction \& Risk Prediction}

\begin{frame}{Shipping Mode Impact on Satisfaction}
\begin{itemize}
    \item Longer shipping times correlate with lower satisfaction and returns.
    \item Optimizing delivery mode selection can improve experience and profitability.
\end{itemize}
\end{frame}

\begin{frame}{Fraud / Return Risk Indicators}
\begin{itemize}
    \item Orders paid via Payment/Debit have higher late + dispute tendencies.
    \item Payment method signals can be used to flag high-risk transactions.
\end{itemize}
\begin{center}
    \includegraphics[width=0.8\textwidth]{ASreport/payment_type_analysis.JPG}
\end{center}
\end{frame}

% Conclusion
\section{Conclusion}

\begin{frame}{Key Takeaways}
\begin{itemize}
    \item Focus growth on high-performing outdoor and apparel product lines.
    \item Improve logistics performance in Europe and Pacific Asia.
    \item Target Consumer and Home Office segments for retention strategies.
    \item Build a predictive model to flag fraud / return risk early.
\end{itemize}
\end{frame}
